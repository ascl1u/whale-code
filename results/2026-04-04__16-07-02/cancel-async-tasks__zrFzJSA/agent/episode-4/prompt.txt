async def test_cancellation_cleanup():
    """Test that cleanup runs even when tasks are cancelled."""
    print("\nTest 3: Cancellation cleanup")
    execution_log.clear()
    cleanup_log.clear()

    async def long_task(task_id: int):
        execution_log.append(f"Task {task_id} started")
        try:
            await asyncio.sleep(10)  # Long sleep
            execution_log.append(f"Task {task_id} completed")
        finally:
            cleanup_log.append(f"Task {task_id} cleanup")

    tasks = [lambda tid=i: long_task(tid) for i in range(5)]

    # Run tasks but cancel after a short time
    async def run_and_cancel():
        task = asyncio.create_task(run_tasks(tasks, max_concurrent=3))
        await asyncio.sleep(0.2)  # Let some tasks start
        task.cancel()
        try:
            await task
        except asyncio.CancelledError:
            pass

    await run_and_cancel()
