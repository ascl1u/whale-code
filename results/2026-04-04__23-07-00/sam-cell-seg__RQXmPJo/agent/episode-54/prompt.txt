def parse_coords(coords_str):
    """Parse comma-separated coordinate string to list of integers."""
    if pd.isna(coords_str) or coords_str == '':
        return []
    return [int(x) for x in str(coords_str).split(',')]

def coords_to_string(coords_list):
    """Convert list of coordinates to comma-separated string."""
    return ','.join([str(int(x)) for x in coords_list])

def get_largest_contour(mask):
    """Extract the largest contiguous contour from a binary mask."""
    contours, _ = cv2.findContours(mask.astype(np.uint8), cv2.RETR_EXTERNAL, cv2.CHAIN_APPROX_SIMPLE)
    if len(contours) == 0:
        return None
    # Get the largest contour
    largest_contour = max(contours, key=cv2.contourArea)
    return largest_contour

def contour_to_coords(contour):
    """Convert contour to separate x and y coordinate lists."""
    if contour is None or len(contour) == 0:
        return [], []

    # Flatten contour and extract x, y coordinates
    contour = contour.squeeze()
    if len(contour.shape) == 1:  # Single point
        contour = contour.reshape(1, -1)

    coords_x = contour[:, 0].tolist()
    coords_y = contour[:, 1].tolist()

    return coords_x, coords_y

def get_bbox_from_coords(xmin, ymin, xmax, ymax):
    """Get bounding box in format expected by SAM."""
    return np.array([xmin, ymin, xmax, ymax])
