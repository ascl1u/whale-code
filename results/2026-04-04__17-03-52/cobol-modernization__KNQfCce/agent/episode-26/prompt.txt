    """Read all account records."""
    accounts = []
    with open(filename, 'rb') as f:
        data = f.read()

    # Each account record is 34 bytes: ID(4) + Name(20) + Balance(10)
    record_size = 34
    for i in range(0, len(data), record_size):

def write_accounts(filename, accounts):
    """Write all account records."""
    with open(filename, 'wb') as f:
        for account in accounts:
            # Format: ID(4) + Name(20) + Balance(10)
            # COBOL PIC 9(10) with REWRITE drops the minus sign for negative numbers
            # It essentially takes the rightmost 10 digits of the string representation
            balance_str = str(account['balance'])
            # Take last 10 characters (drops minus sign if present)
            balance_formatted = balance_str[-10:].zfill(10)
            record = (
                account['id'].encode('ascii') +
                account['name'].encode('ascii') +
                balance_formatted.encode('ascii')
            )
            f.write(record)
