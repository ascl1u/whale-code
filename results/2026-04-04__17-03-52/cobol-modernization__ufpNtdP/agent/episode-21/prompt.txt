def read_books(filename):
    """Read all book records from BOOKS.DAT."""
    books = []
    with open(filename, 'rb') as f:
        data = f.read()

    # Each book record is 28 bytes: 4 (ID) + 20 (Title) + 4 (Owner)
    record_size = 28
    for i in range(0, len(data), record_size):
        if i + record_size <= len(data):
            record = data[i:i+record_size]
            book_id = record[0:4].decode('ascii')
            book_title = record[4:24].decode('ascii')
            book_owner = record[24:28].decode('ascii')
            books.append({
                'id': book_id,
                'title': book_title,
                'owner': book_owner
            })

    return books

def write_books(filename, books):
    """Write all book records to BOOKS.DAT."""
    with open(filename, 'wb') as f:
        for book in books:
            record = (
                book['id'].encode('ascii') +
                book['title'].encode('ascii') +
                book['owner'].encode('ascii')
            )
            f.write(record)
