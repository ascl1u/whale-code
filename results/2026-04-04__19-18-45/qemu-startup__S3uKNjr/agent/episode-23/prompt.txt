import struct

def parse_directory_record(data, offset):
    if offset >= len(data) or data[offset] == 0:
        return None
    length = data[offset]
    if length == 0:
        return None
    lba = struct.unpack('<I', data[offset+2:offset+6])[0]
    size = struct.unpack('<I', data[offset+10:offset+14])[0]
    flags = data[offset+25]
    name_len = data[offset+32]
    name = data[offset+33:offset+33+name_len].decode('ascii', errors='ignore')
    return {'length': length, 'lba': lba, 'size': size, 'flags': flags, 'name': name, 'is_dir': (flags & 2) != 0}
